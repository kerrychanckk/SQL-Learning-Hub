Database Schema

students

  • student_id INTEGER PRIMARY KEY
  • class TEXT
  • no INTEGER
  • last_name TEXT

  • gender TEXT
  • dob DATE
  • district TEXT

subjects

  • subject_code TEXT PK
  • subject_name TEXT

exams

  • student_id INTEGER
  • subject_code TEXT FK
  • score INTEGER
